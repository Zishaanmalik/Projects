function predict() { const data = { WindSpeed: parseFloat(document.getElementById("ws").value), WindDirection: parseFloat(document.getElementById("wd").value), Temperature: parseFloat(document.getElementById("temp").value), Pressure: parseFloat(document.getElementById("press").value), AirDensity: parseFloat(document.getElementById("density").value), RotorRPM: parseFloat(document.getElementById("rpm").value), }; // Check if any input is empty if (Object.values(data).some(isNaN)) { document.getElementById("result").innerHTML = "❌ Please enter all values!"; return; } fetch("http://127.0.0.1:5000/predict", { method: "POST", headers: {"Content-Type": "application/json"}, body: JSON.stringify(data) }) .then(response => response.json()) .then(result => { if (result.prediction !== undefined) { document.getElementById("result").innerHTML = `⚡ Predicted Wind Power: ${result.prediction.toFixed(2)} kW`; } else { document.getElementById("result").innerHTML = "❌ Error: " + result.error; } }) .catch(err => { document.getElementById("result").innerHTML = "❌ Connection Error!"; }); }
